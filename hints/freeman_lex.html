Here is the whole solution:

from collections.abc import Iterable


def string_permutations(s: str) -> Iterable[str]:
    
    if len(s) < 2: return [s]
    all = []
    for i, char in enumerate(s):
        remaining_chars = s[:i] + s[i + 1:]
        for perm in string_permutations(remaining_chars):
            all.append(char + perm)
    
    return sorted(all)

Use standard module itertools and permutations function to generate all variants, join each into single string (as each of them is a tuple) with map function and return sorted list. 

from collections.abc import Iterable
from itertools import permutations


def string_permutations(s: str) -> Iterable[str]:
    
    return sorted(map("".join, permutations(s)))
