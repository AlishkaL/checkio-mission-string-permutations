You just need to join all previous hints, return sorted general list and wrap your code into function expression. Using type hints for input and output data are highly recommended.

Here is the whole solution:

function stringPermutations(s: string): string[] {

    if (s.length <= 1) return [s];
    const all = [];
    for (let i = 0; i < s.length; i++) {
        const char = s[i];
        const remainingChars = s.slice(0, i) + s.slice(i + 1);
        for (const perm of stringPermutations(remainingChars)) {
            all.push(char + perm);
        }
    }
    return all.sort();
}
